drop database negozio;

create database negozio;

use negozio;

create table marche
(
   id_marca int AUTO_INCREMENT,
   nome varchar(30) not null,
   primary key (id_marca)
) ENGINE=MyISAM;

create table categorie
(
   id_categoria int AUTO_INCREMENT,
   nome varchar(30) not null,
   primary key (id_categoria)
) ENGINE=MyISAM;

create table tipi_prodotto
(
   id_tipo int AUTO_INCREMENT,
   nome varchar(30) not null,
   primary key (id_tipo),
   fk_categoria int,
   foreign key (fk_categoria) references categorie(id_categoria)
) ENGINE=MyISAM;


create table prodotti
(
   id_prodotto int AUTO_INCREMENT,
   nome varchar(60) not null,
   descrizione varchar(256),
   prezzo double not null,
   FULLTEXT (nome),
   primary key (id_prodotto),
   fk_marca int,
   foreign key (fk_marca) references marche(id_marca),
   fk_tipo int,
   foreign key (fk_tipo) references tipi_prodotto(id_tipo)
) ENGINE=MyISAM;


create table utenti
(
    codice_fiscale char(16),
    nome varchar(30) not null,
    cognome varchar(30) not null,
    data_nascita date,
    indirizzo varchar(100),
    citta varchar(50),
    CAP varchar(10),
    email varchar(40) not null,
    password varchar(40) not null,
    numero_telefono varchar(15),
    tipo enum('amministratore','operatore','cliente') not null,
    primary key(codice_fiscale)
) ENGINE=MyISAM;

create table ordini
(
   id_ordine int AUTO_INCREMENT,
   data_ordine date,
   indirizzo varchar(100),
   citta varchar(50),
   CAP varchar(10),
   importo_totale double not null,
   fk_codice_fiscale char(16),
   foreign key (fk_codice_fiscale) references utenti(codice_fiscale),
   primary key (id_ordine)
) ENGINE=MyISAM;

create table contenere
(
   fk_prodotto int,
   foreign key (fk_prodotto) references prodotti(id_prodotto),
   fk_ordine int,
   foreign key (fk_ordine) references ordini(id_ordine),
   primary key (fk_prodotto, fk_ordine)
) ENGINE=MyISAM;


create user 'operatore'@'localhost' IDENTIFIED BY 'password';
GRANT SELECT ON negozio.* TO 'operatore'@'localhost';
GRANT INSERT, UPDATE ON negozio.prodotti TO 'operatore'@'localhost';
GRANT INSERT, UPDATE ON negozio.tipi_prodotto TO 'operatore'@'localhost';
GRANT INSERT, UPDATE ON negozio.marche TO 'operatore'@'localhost';
GRANT INSERT, UPDATE ON negozio.categorie TO 'operatore'@'localhost';
